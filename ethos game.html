    <script>

        // --- Data Arrays ---

        const questionStarters = [

            "Imagine if", "What would happen if", "Describe a world where", "What's the secret story behind",

            "If you could ask", "Pretend that", "What funny reason explains why", "Suppose",

            "Invent a holiday celebrating", "What if", "Tell a tall tale about"

        ];

        const subjects = [

            "dinosaurs", "garden gnomes", "talking squirrels", "lost socks", "clouds",

            "friendly monsters under the bed", "pet rocks", "fairies", "a mischievous teapot",

            "bicycles", "books", "stars", "singing flowers", "tiny dragons", "your reflection"

        ];

        const contexts = [

            "learned to skateboard?", "started a band?", "could grant wishes, but only silly ones?",

            "secretly ran the internet?", "threw a surprise party?", "decided to become astronauts?",

            "wore tiny hats?", "could only communicate through riddles?", "were afraid of butterflies?",

            "built elaborate sandcastles overnight?", "could change color based on their mood?",

            "held weekly meetings in your attic?", "tried to learn magic tricks?", "went on vacation to the moon?"

        ];

        const twists = [

            "in a world where everything is made of jelly.",

            "during a never-ending dance party.",

            "while everyone speaks only in rhymes.",

            "and gravity suddenly reversed.",

            "with a mysterious talking cheese guiding them.",

            "on a floating island shaped like a banana.",

            "with time moving backwards.",

            "underwater with disco lights.",

            "while dreaming inside a toaster.",

            "on Mars, but nobody knows it yet."

        ];


        // --- Element References ---

        const reel1 = document.getElementById('reel1');

        const reel2 = document.getElementById('reel2');

        const reel3 = document.getElementById('reel3');

        const reel4 = document.getElementById('reel4');


        const spinButton = document.getElementById('spinButton');

        const messageArea = document.getElementById('messageArea'); // For user feedback


        const rerollButton1 = document.getElementById('reroll1');

        const rerollButton2 = document.getElementById('reroll2');

        const rerollButton3 = document.getElementById('reroll3');

        const rerollButton4 = document.getElementById('reroll4');


        const surpriseButton = document.getElementById('surpriseButton');

        const sendToGeminiButton = document.getElementById('sendToGeminiButton');


        // --- Store current selections and the generated question ---

        let currentStarter = "Imagine if..."; // Initial placeholder text

        let currentSubject = "A certain thing...";

        let currentContext = "Had a funny context...";

        let currentTwist = "With a twist...";

        let generatedQuestion = ""; // Stores the fully formed question


        // --- Helper Function: Get Random Item ---

        function getRandomItem(arr) {

            const randomIndex = Math.floor(Math.random() * arr.length);

            return arr[randomIndex];

        }


        // --- Helper Function: Animate Reel ---

        function animateReel(reelElement) {

            reelElement.classList.add('spinning');

            setTimeout(() => {

                reelElement.classList.remove('spinning');

            }, 500);

        }


        // --- Function to update the internal generated question string ---

        function updateGeneratedQuestion(isSurprise = false) {

            clearMessage(); // Clear any previous messages

            if (isSurprise) {

                generatedQuestion = `If ${currentSubject.toLowerCase()} ${currentContext.toLowerCase().replace('?', '')} ${currentTwist.toLowerCase().replace('.', '')}, what would ${currentStarter.toLowerCase()} be?`;

            } else {

                generatedQuestion = `${currentStarter} ${currentSubject.toLowerCase()} ${currentContext.toLowerCase()} ${currentTwist.toLowerCase()}`;

            }

        }


        // --- Display messages to the user ---

        function displayMessage(text) {

            if (messageArea) {

                messageArea.textContent = text;

            }

        }

        function clearMessage() {

            if (messageArea) {

                messageArea.textContent = '';

            }

        }



        // --- Spin All Reels Function ---

        function spinAllReels() {

            currentStarter = getRandomItem(questionStarters);

            currentSubject = getRandomItem(subjects);

            currentContext = getRandomItem(contexts);

            currentTwist = getRandomItem(twists);


            reel1.textContent = currentStarter;

            animateReel(reel1);

            reel2.textContent = currentSubject;

            animateReel(reel2);

            reel3.textContent = currentContext;

            animateReel(reel3);

            reel4.textContent = currentTwist;

            animateReel(reel4);


            updateGeneratedQuestion();

            displayMessage("New question ready! Try sending it to Gemini.");

        }


        // --- Reroll Single Reel Function ---

        function rerollReel(reelNumber) {

            switch (reelNumber) {

                case 1:

                    currentStarter = getRandomItem(questionStarters);

                    reel1.textContent = currentStarter;

                    animateReel(reel1);

                    break;

                case 2:

                    currentSubject = getRandomItem(subjects);

                    reel2.textContent = currentSubject;

                    animateReel(reel2);

                    break;

                case 3:

                    currentContext = getRandomItem(contexts);

                    reel3.textContent = currentContext;

                    animateReel(reel3);

                    break;

                case 4:

                    currentTwist = getRandomItem(twists);

                    reel4.textContent = currentTwist;

                    animateReel(reel4);

                    break;

            }

            updateGeneratedQuestion(); // Update the internal question string

            displayMessage("Reel rerolled! New question ready.");

        }


        // --- Surprise Me Function ---

        function generateSurprise() {

            currentStarter = getRandomItem(questionStarters);

            currentSubject = getRandomItem(subjects);

            currentContext = getRandomItem(contexts);

            currentTwist = getRandomItem(twists);


            reel1.textContent = currentStarter;

            animateReel(reel1);

            reel2.textContent = currentSubject;

            animateReel(reel2);

            reel3.textContent = currentContext;

            animateReel(reel3);

            reel4.textContent = currentTwist;

            animateReel(reel4);


            updateGeneratedQuestion(true); // Use surprise format for the internal question

            displayMessage("Surprise question generated! Send it to Gemini!");

        }


        // --- Send to Gemini Function ---

        function sendToGemini() {

            if (generatedQuestion) {

                const encodedQuestion = encodeURIComponent(generatedQuestion);

                const geminiUrl = `https://gemini.google.com/app?prompt=${encodedQuestion}`;

                window.open(geminiUrl, '_blank');

                clearMessage(); // Clear message after sending

            } else {

                displayMessage("Spin the reels or hit 'Surprise Me!' to generate a question first!");

            }

        }


        // --- Event Listeners ---

        spinButton.addEventListener('click', spinAllReels);


        rerollButton1.addEventListener('click', () => rerollReel(1));

        rerollButton2.addEventListener('click', () => rerollReel(2));

        rerollButton3.addEventListener('click', () => rerollReel(3));

        rerollButton4.addEventListener('click', () => rerollReel(4));


        surpriseButton.addEventListener('click', generateSurprise);

        sendToGeminiButton.addEventListener('click', sendToGemini);


        // --- Initial State ---

        // Set initial text for reels

        reel1.textContent = currentStarter;

        reel2.textContent = currentSubject;

        reel3.textContent = currentContext;

        reel4.textContent = currentTwist;

        // spinAllReels(); // Optionally spin on load to have a question ready


    </script>
